#include <stdio.h>
long long dpmemo[101][101];
void getMax(int N, int P) {
	for (int i = 1; i <= N; i++) {
		dpmemo[1][i] = i;
	}
	for (int i = 2; i <= P; i++) {
		for (int j = i; j <= N; j++) {
			for (int k = 1; k < j; k++) {
				if (dpmemo[i][j] < dpmemo[i - 1][j - k] * k) {
					dpmemo[i][j] = dpmemo[i - 1][j - k] * k;
				}
			}
		}
	}
}

int main() {

	int tc;
	scanf("%d", &tc);
	getMax(100, 100);
	for (int testcase = 1; testcase <= tc; testcase++) {
		int N, P;
		scanf("%d %d", &N, &P);
#if 0

#endif

}
